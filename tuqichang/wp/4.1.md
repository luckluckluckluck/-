    # 读取文件原始字节
    with open(input_path, "rb") as f:
        data = f.read()
    
    # 字节倒序（核心一行）
    reversed_data = data[::-1]
    
    # 写入新文件
    with open(output_path, "wb") as f:
        f.write(reversed_data)
    
    print(f"✅ 倒序完成！")
